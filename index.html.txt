    // ---------- 华文生字库 ----------
    const questionsDB = [

        { word: "学习", pinyin: "xué xí", meaning: "获取知识", options: ["玩耍", "学习", "睡觉", "跑步"], correct: "学习", sentence: "我们要认真___。" },
        { word: "朋友", pinyin: "péng yǒu", meaning: "关系好的人", options: ["敌人", "陌生人", "朋友", "老师"], correct: "朋友", sentence: "他是我的好___。" }

    let currentQuestionObj = null;

    let canAnswer = true;

    let currentSkin = "😊";

    // 排行榜(本地存储)
    let leaderboard = JSON.parse(localStorage.getItem("coinLeaderboard")) || [];
    function updateLeaderboard(name, score) {
        leaderboard.push({ name: name || "玩家", score: score });
        leaderboard.sort((a,b)=>b.score - a.score);
        if(leaderboard.length > 8) leaderboard.pop();
        localStorage.setItem("coinLeaderboard", JSON.stringify(leaderboard));
        renderLeaderboard();
    }
    function renderLeaderboard() {
        const rankListDiv = document.querySelector(".rank-list");
        if(rankListDiv){
            rankListDiv.innerHTML = leaderboard.map((p,idx)=>`<li>${idx+1}. ${p.name} 🪙 ${p.score}</li>`).join('');
        }else{
            // 如果不存在动态创建排行榜面板
            let panel = document.querySelector(".rank-panel");
            if(!panel){
                panel = document.createElement('div');
                panel.className = "rank-panel";
                document.body.appendChild(panel);
            }
            panel.innerHTML = `<div class="rank-title">🏆 金币富豪榜</div><ul class="rank-list">${leaderboard.map((p,idx)=>`<li>${idx+1}. ${p.name} 🪙 ${p.score}</li>`).join('')}</ul>`;
        }
    }
    function addToLeaderboardFromCurrent() { updateLeaderboard("勇者", playerTotalScore); }

    // 金币特效
    const canvas = document.getElementById('coinCanvas');
    let ctx = canvas.getContext('2d');
    function resizeCanvas(){ canvas.width = window.innerWidth; canvas.height = window.innerHeight; }
    window.addEventListener('resize', resizeCanvas); resizeCanvas();
    let coinsParticles = [];
    function createCoinBurst(x, y) {
        for(let i=0;i<24;i++){
            const angle = Math.random() * Math.PI * 2;
            const speed = 3 + Math.random() * 7;
            const vx = Math.cos(angle) * speed;
            const vy = Math.sin(angle) * speed - 3;
            coinsParticles.push({ x, y, vx, vy, life: 1, size: 24+Math.random()*12, alpha:1 });
        }
        playSound('coin');
    }
    function drawCoins(){
        if(!ctx) return;
        ctx.clearRect(0,0,canvas.width,canvas.height);
        for(let i=0;i<coinsParticles.length;i++){
            let p = coinsParticles[i];
            ctx.save();
            ctx.globalAlpha = p.life;
            ctx.font = `${p.size}px "Segoe UI Emoji"`;
            ctx.fillStyle = "#FFD966";
            ctx.shadowBlur = 8;
            ctx.fillText("🪙", p.x, p.y);
            ctx.restore();
            p.x += p.vx;
            p.y += p.vy;
            p.life -= 0.02;
        }
        coinsParticles = coinsParticles.filter(p=>p.life>0);
        requestAnimationFrame(drawCoins);
    }
    drawCoins();

    // 简易音效 (使用Web Audio)
    function playSound(type){
        let audioCtx = new (window.AudioContext || window.webkitAudioContext)();
        let osc = audioCtx.createOscillator();
        let gain = audioCtx.createGain();
        osc.connect(gain);
        gain.connect(audioCtx.destination);
        if(type === 'coin'){
            osc.frequency.value = 880; gain.gain.value = 0.2; osc.type = 'sine';
            osc.start(); gain.gain.exponentialRampToValueAtTime(0.00001, audioCtx.currentTime+0.3);
            osc.stop(audioCtx.currentTime+0.3);
        }else if(type === 'correct'){
            osc.frequency.value = 523.25; gain.gain.value = 0.15;
            osc.start(); gain.gain.exponentialRampToValueAtTime(0.00001, audioCtx.currentTime+0.2);
            osc.stop(audioCtx.currentTime+0.2);
        }else if(type === 'wrong'){
            osc.frequency.value = 220; gain.gain.value = 0.2;
            osc.start(); gain.gain.exponentialRampToValueAtTime(0.00001, audioCtx.currentTime+0.3);
            osc.stop(audioCtx.currentTime+0.3);
        }else if(type === 'buy'){
            osc.frequency.value = 659.25; gain.gain.value = 0.1;
            osc.start(); gain.gain.exponentialRampToValueAtTime(0.00001, audioCtx.currentTime+0.15);
            osc.stop(audioCtx.currentTime+0.15);
        }
    }

    function showToast(msg, isError=false){
        let div = document.createElement('div');
        div.className = 'toast-msg';
        div.innerText = msg;
        if(isError) div.style.background = '#711010';
        document.body.appendChild(div);
        setTimeout(()=>div.remove(), 1000);
    }

    function updateUI(){
        document.getElementById('coinAmount').innerText = playerCoins;
        document.getElementById('totalScore').innerText = playerTotalScore;
        document.getElementById('marioFace').innerText = currentSkin;
        if(isMultiplayer){
            document.getElementById('multiStatusMsg').innerHTML = `⚔️ 多人模式 | 你的金币:${playerCoins} 🆚 对手金币:${multiOpponentScore} ⚔️`;
        }else{
            document.getElementById('multiStatusMsg').innerHTML = `⭐ 单人模式·挑战华文 ⭐ 分数:${playerTotalScore}`;
        }
    }

    function stopTimer(){ if(timerInterval){ clearInterval(timerInterval); timerInterval=null; } }
    function startTimer(){
        if(timerInterval) stopTimer();
        timerInterval = setInterval(()=>{
            if(!canAnswer) return;
            if(currentTime <= 1){
                stopTimer();
                handleTimeout();
            }else{
                currentTime--;
                let timerSpan = document.getElementById('timerNum');
                timerSpan.innerText = currentTime;
                if(currentTime <= 3) timerSpan.style.color = "red";
                else timerSpan.style.color = "white";
            }
        },1000);
    }
    function handleTimeout(){
        if(!canAnswer) return;
        canAnswer = false;
        showToast("⏰ 超时！正确答案是: "+currentCorrect);
        playSound('wrong');
        document.querySelector('.quiz-area').classList.add('wrong-shake');
        setTimeout(()=>document.querySelector('.quiz-area').classList.remove('wrong-shake'),300);
        loadNewQuestion();
    }
    function loadNewQuestion(){
        stopTimer();
        currentTime = 10;
        document.getElementById('timerNum').innerText = "10";
        document.getElementById('timerNum').style.color = "white";
        canAnswer = true;
        let randomIndex = Math.floor(Math.random() * questionsDB.length);
        currentQuestionObj = {...questionsDB[randomIndex]};
        let q = currentQuestionObj;
        let sentenceDisplay = q.sentence.replace("___", "______");
        document.getElementById('questionText').innerHTML = `📖 ${q.word} (${q.pinyin})<br>✏️ ${sentenceDisplay}`;
        currentCorrect = q.correct;
        let opts = [...q.options];
        for(let i=opts.length;i<4;i++) opts.push("?");
        let optionsDiv = document.getElementById('optionsContainer');
        optionsDiv.innerHTML = '';
        opts.forEach(opt=>{
            let btn = document.createElement('button');
            btn.className = 'option-btn';
            btn.innerText = opt;
            btn.onclick = ()=> handleAnswer(opt);
            optionsDiv.appendChild(btn);
        });
        startTimer();
    }
    function handleAnswer(selected){
        if(!canAnswer) return;
        canAnswer = false;
        stopTimer();
        let isCorrect = (selected === currentCorrect);
        if(isCorrect){
            let reward = 10;
            if(doubleModeActive){ reward*=2; doubleModeActive=false; showToast("✨双倍得分生效✨",false);}
            playerCoins += reward;
            playerTotalScore += reward;
            if(isMultiplayer){
                multiPlayerScore += reward;
                playerCoins = multiPlayerScore;
                playerTotalScore = multiPlayerScore;
            }
            updateUI();
            playSound('correct');
            showToast(`✅ +${reward}金币！`,false);
            // 金币爆炸特效
            let rect = document.querySelector('.quiz-area').getBoundingClientRect();
            createCoinBurst(rect.left+rect.width/2, rect.top+50);
            // 表情微笑/开心
            currentSkin = "😄🎉";
            document.getElementById('marioFace').innerText = currentSkin;
            setTimeout(()=>{ currentSkin = "😊"; updateUI(); }, 800);
            // 多人模式累加对手AI随意加金币
            if(isMultiplayer && Math.random()>0.5){
                let aiGain = 5+Math.floor(Math.random()*8);
                multiOpponentScore += aiGain;
                showToast(`对手获得 ${aiGain} 金币！`,false);
                updateUI();
            }
        }else{
            playSound('wrong');
            showToast(`❌ 错误! 答案是 ${currentCorrect}`, true);
            document.querySelector('.quiz-area').classList.add('wrong-shake');
            setTimeout(()=>document.querySelector('.quiz-area').classList.remove('wrong-shake'),400);
            if(isMultiplayer){
                // 对手加一点分惩罚
                multiOpponentScore += 5;
                updateUI();
            }
        }
        if(isMultiplayer){
            playerCoins = multiPlayerScore;
            playerTotalScore = multiPlayerScore;
        }
        updateUI();
        loadNewQuestion();
    }

    // 商店道具购买
    function buyItem(item){
        if(item === 'time' && playerCoins >=20){
            playerCoins-=20; playSound('buy'); if(canAnswer) { currentTime +=5; document.getElementById('timerNum').innerText = currentTime; showToast("⏱️ +5秒！"); }
            else showToast("当前无法加时");
        }else if(item === 'hint' && playerCoins >=30){
            playerCoins-=30; playSound('buy');
            let btns = document.querySelectorAll('.option-btn');
            for(let btn of btns){
                if(btn.innerText !== currentCorrect && btns.length>2){
                    btn.style.display = 'none';
                    setTimeout(()=>{btns.forEach(b=>b.style.display='flex');},3000);
                    break;
                }
            }
            showToast("💡 已帮你排除一个错误选项");
        }else if(item === 'double' && playerCoins >=40){
            playerCoins-=40; doubleModeActive=true; playSound('buy'); showToast("✨ 下一题双倍得分!");
        }else if(item === 'skin' && playerCoins >=15){
            playerCoins-=15; currentSkin = "🐱🎩"; playSound('buy'); updateUI(); showToast("🎭 新皮肤: 魔法猫!");
        }else{ showToast("金币不足!",true); return false;}
        updateUI();
        return true;
    }

    // 多人模式对战逻辑
    function startMultiBattle(){

        isMultiplayer = true;
        multiPlayerScore = playerCoins;
        multiOpponentScore = 50 + Math.floor(Math.random()*40);

        playerTotalScore = multiPlayerScore;

        loadNewQuestion();
    }

        if(!isMultiplayer) startMultiBattle();
